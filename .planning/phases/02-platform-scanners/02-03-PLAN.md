```python
# mis/scanners/hotmart.py — httpx SSR, igual ao KiwifyScanner
import httpx
from bs4 import BeautifulSoup
import re

HOTMART_BASE_URL = "https://hotmart.com/pt-br/marketplace/produtos"
HOTMART_PLATFORM_ID = None  # verificar: SELECT id FROM platforms WHERE name LIKE '%Hotmart%'

SELECTORS_ORDERED = [
    ("a.product-link", "aria-label"),          # primary — confirmado ao vivo
    (".product-card-alt a[aria-label]", "aria-label"),  # fallback 1
]

async def _fetch_hotmart_page(category_slug: str) -> str:
    """Fetch SSR HTML via httpx. Returns HTML string."""
    url = f"{HOTMART_BASE_URL}?category={category_slug}"
    async with httpx.AsyncClient(follow_redirects=True, timeout=30) as client:
        resp = await client.get(url, headers={"Accept-Language": "pt-BR,pt;q=0.9"})
        resp.raise_for_status()
        return resp.text

def _parse_products(html: str, niche_id: int, rank_offset: int = 0) -> list:
    """Parse product cards from SSR HTML."""
    soup = BeautifulSoup(html, "lxml")
    for selector, label_attr in SELECTORS_ORDERED:
        cards = soup.select(selector)
        if cards:
            products = []
            for rank, card in enumerate(cards, start=rank_offset + 1):
                href = card.get("href", "")
                aria = card.get("aria-label", "")
                # external_id: código alfanumérico no final da URL (ex: E45853768C)
                code_match = re.search(r'/([A-Z][0-9A-Z]{6,10})\?', href)
                external_id = code_match.group(1) if code_match else href.split("/")[-1]
                title = aria.split(" - ")[0].strip() if " - " in aria else aria.strip()
                products.append(Product(
                    external_id=external_id, title=title, url=href,
                    platform_id=HOTMART_PLATFORM_ID, niche_id=niche_id, rank=rank,
                ))
            return products
    # todos os seletores falharam
    return []
```

    ```python
    def register_scanner_jobs(config: dict) -> None:
        """Register one APScheduler job per platform scanner. Call before start_scheduler()."""
        from .scanners.hotmart import run_hotmart_scan
        from .scanners.clickbank import run_clickbank_scan
        from .scanners.kiwify import run_kiwify_scan

        scheduler = get_scheduler()
        scan_schedule = config.get("settings", {}).get("scan_schedule", "0 3 * * *")
        trigger = CronTrigger.from_crontab(scan_schedule)

        for job_id, func in [
            ("scanner_hotmart", run_hotmart_scan),
            ("scanner_clickbank", run_clickbank_scan),
            ("scanner_kiwify", run_kiwify_scan),
        ]:
            scheduler.add_job(func, trigger, args=[config], id=job_id, replace_existing=True)
            log.info("scanner.job.registered", job_id=job_id, schedule=scan_schedule)
    ```
