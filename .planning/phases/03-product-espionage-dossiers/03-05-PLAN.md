    Criar mis/spy_orchestrator.py com funções públicas run_spy, run_spy_url, run_spy_batch:

    ```python
    import asyncio, json
    from datetime import datetime
    from itertools import count
    from typing import Optional
    import structlog
    from .config import get_config
    from .db import get_db
    from .spies.sales_page import SalesPageScraper
    from .spies.meta_ads import MetaAdsScraper
    from .spies.reviews import ReviewsScraper
    from .spies.completeness_gate import SpyData, check_completeness
    from .intelligence.copy_analyzer import analyze_copy, CopyAnalysisError
    from .intelligence.dossier_generator import generate_dossier, DossierGenerationError
    from .exceptions import ScraperError

    log = structlog.get_logger()
    _spy_counter = count()

    async def run_spy(product_id: int, force: bool = False) -> None:
        """Espiona produto por ID. Se já tem dossier done e force=False, pula."""
        config = get_config()
        db = get_db(config.db_path)
        product = dict(db["products"].get(product_id))

        # Verificar se já tem dossier (skip se automático, respy se force=True)
        existing = list(db["dossiers"].rows_where("product_id = ?", [product_id]))
        if existing and existing[0]["status"] == "done" and not force:
            log.info("spy.skipped", product_id=product_id, reason="dossier_exists")
            return

        # Criar/atualizar registro dossier como pending
        dossier_id = _upsert_dossier_status(db, product_id, "running")

        try:
            await _execute_spy_pipeline(product, dossier_id, config, db)
        except Exception as e:
            log.error("spy.failed", alert="spy_failed", product_id=product_id, error=str(e))
            _upsert_dossier_status(db, product_id, "failed", dossier_id=dossier_id)

    async def run_spy_url(url: str) -> None:
        """Espiona produto por URL arbitrária (platform-agnostic). Sempre re-espiona."""
        config = get_config()
        db = get_db(config.db_path)

        # Criar produto mínimo no banco para ter product_id
        product = {"url": url, "title": "", "platform_id": 0, "external_id": url, "rank": 999}
        product_id = db["products"].insert(product, ignore=True).last_pk

        dossier_id = _upsert_dossier_status(db, product_id, "running")
        try:
            await _execute_spy_pipeline(product, dossier_id, config, db)
        except Exception as e:
            log.error("spy.failed", alert="spy_failed", product_url=url, error=str(e))
            _upsert_dossier_status(db, product_id, "failed", dossier_id=dossier_id)

    async def run_spy_batch(products: list[dict], max_concurrent: int = 3) -> None:
        """Executa espionagem em batch com concorrência limitada. Manuais têm prioridade."""
        semaphore = asyncio.Semaphore(max_concurrent)
        queue = asyncio.PriorityQueue()

        for p in products:
            priority = 0 if p.get("is_manual") else p.get("rank", 999)
            await queue.put((priority, next(_spy_counter), p["id"]))

        async def worker():
            while not queue.empty():
                _, _, product_id = await queue.get()
                async with semaphore:
                    await run_spy(product_id)
                queue.task_done()

        await asyncio.gather(*[worker() for _ in range(max_concurrent)])

    async def _execute_spy_pipeline(product: dict, dossier_id: int, config, db) -> None:
        min_reviews = getattr(config.spy, "min_reviews", 10)

        # 1. Coletar dados (todos em paralelo para eficiência)
        async with SalesPageScraper() as sp_scraper:
            extracted = await sp_scraper.extract(product["url"])

        meta_scraper = MetaAdsScraper()
        ads = await meta_scraper.fetch_ads(product.get("title", ""))

        async with ReviewsScraper() as rev_scraper:
            reviews = await rev_scraper.collect(
                platform=_platform_slug(product.get("platform_id", 0)),
                product_url=product["url"],
                product_title=product.get("title", ""),
            )

        # 2. Montar SpyData
        spy_data = SpyData(
            copy_text=extracted.get("headlines", [""])[0] + " " + " ".join(extracted.get("arguments", [])),
            offer_data={k: extracted[k] for k in ("price","bonuses","guarantees","upsells","downsells") if k in extracted},
            reviews=reviews,
            ads=ads,
        )
        # Nota: copy_text completo = todas as seções da copy concatenadas
        full_copy = "\n".join([
            " ".join(extracted.get("headlines", [])),
            " ".join(extracted.get("sub_headlines", [])),
            " ".join(extracted.get("arguments", [])),
            extracted.get("narrative_structure", ""),
        ])
        spy_data.copy_text = full_copy

        # 3. Gate de completude
        gate_passed, confidence = check_completeness(spy_data, min_reviews=min_reviews)

        if not spy_data.copy_text or len(spy_data.copy_text) < 100:
            raise ScraperError("Copy ausente — não é possível gerar dossiê")

        # 4. Pipeline LLM (sempre roda se copy ok — gate_passed=False gera dossier parcial)
        copy_analysis = await analyze_copy(spy_data)
        dossier_data = await generate_dossier(spy_data, copy_analysis, dossier_id, config.db_path)

        # Incorporar copy_analysis e confidence no dossier final
        dossier_data["copy_analysis"] = copy_analysis
        dossier_data["confidence_score"] = confidence
        dossier_data["incomplete"] = not gate_passed

        # 5. Persistir
        db["dossiers"].update(dossier_id, {
            "status": "done",
            "dossier_json": json.dumps(dossier_data, ensure_ascii=False),
            "confidence_score": confidence,
            "incomplete": int(not gate_passed),
            "updated_at": datetime.utcnow().isoformat(),
        })
        log.info("spy.done", dossier_id=dossier_id, confidence=confidence, incomplete=not gate_passed)

    def _upsert_dossier_status(db, product_id: int, status: str, dossier_id: Optional[int] = None) -> int:
        if dossier_id:
            db["dossiers"].update(dossier_id, {"status": status, "updated_at": datetime.utcnow().isoformat()})
            return dossier_id
        existing = list(db["dossiers"].rows_where("product_id = ?", [product_id]))
        if existing:
            db["dossiers"].update(existing[0]["id"], {"status": status})
            return existing[0]["id"]
        return db["dossiers"].insert({"product_id": product_id, "status": status, "created_at": datetime.utcnow().isoformat()}).last_pk

    def _platform_slug(platform_id: int) -> str:
        return {1: "hotmart", 2: "clickbank", 3: "kiwify"}.get(platform_id, "unknown")
    ```
