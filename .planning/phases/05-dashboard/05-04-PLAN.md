    ```python
    def check_and_create_alert_if_top20_entry(
        db: sqlite_utils.Database,
        db_path: str,
        product: Product,
        new_rank: int,
        threshold: int = 20,
    ) -> None:
        """Create an alert when a product enters the top-N ranking for the first time.

        Logic:
          - If product already exists in DB with rank <= threshold → no alert (already in top N)
          - If product does not exist in DB (new) AND new_rank <= threshold → create alert
          - If product exists with rank > threshold AND new_rank <= threshold → create alert
          - Otherwise → no alert
        """
        if new_rank > threshold:
            return  # Fast path: not in top N, no alert needed

        existing = list(db.execute(
            "SELECT rank FROM products WHERE platform_id = ? AND external_id = ?",
            [product.platform_id, product.external_id]
        ))

        if existing:
            old_rank = existing[0][0]
            if old_rank is not None and old_rank <= threshold:
                return  # Already in top N — not a new entry

        # New entry into top N: create alert
        # Get platform slug from DB
        platforms = list(db.execute(
            "SELECT slug FROM platforms WHERE id = ?", [product.platform_id]
        ))
        platform_slug = platforms[0][0] if platforms else str(product.platform_id)

        # Get niche slug from DB
        niches = list(db.execute(
            "SELECT slug FROM niches WHERE id = ?", [product.niche_id]
        ))
        niche_slug = niches[0][0] if niches else str(product.niche_id)

        from mis.alert_repository import create_alert
        create_alert(
            db_path=db_path,
            product_id=product.platform_id,  # NOTE: precisa do products.id real
            platform_slug=platform_slug,
            niche_slug=niche_slug,
            position=new_rank,
        )
    ```

    ```python
    def save_batch_with_alerts(db: sqlite_utils.Database, db_path: str, products: list[Product]) -> None:
        from mis.product_repository import save_batch
        # 1. Capturar ranks atuais ANTES do upsert
        old_ranks = {}
        for p in products:
            rows = list(db.execute(
                "SELECT rank FROM products WHERE platform_id=? AND external_id=?",
                [p.platform_id, p.external_id]
            ))
            old_ranks[(p.platform_id, p.external_id)] = rows[0][0] if rows else None

        # 2. Upsert
        save_batch(db, products)

        # 3. Criar alertas para novas entradas no top 20
        THRESHOLD = 20
        for p in products:
            if p.rank > THRESHOLD:
                continue
            old_rank = old_ranks.get((p.platform_id, p.external_id))
            if old_rank is not None and old_rank <= THRESHOLD:
                continue  # Já estava no top 20
            # Nova entrada: buscar products.id real pós-upsert
            rows = list(db.execute(
                "SELECT id FROM products WHERE platform_id=? AND external_id=?",
                [p.platform_id, p.external_id]
            ))
            if not rows:
                continue
            real_id = rows[0][0]
            platforms = list(db.execute("SELECT slug FROM platforms WHERE id=?", [p.platform_id]))
            platform_slug = platforms[0][0] if platforms else str(p.platform_id)
            niches = list(db.execute("SELECT slug FROM niches WHERE id=?", [p.niche_id]))

            from mis.alert_repository import create_alert

                         platform_slug=platform_slug, niche_slug=niche_slug, position=p.rank)
    ```
