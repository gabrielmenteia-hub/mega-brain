    ```python
    import os, httpx, structlog
    from ..exceptions import ScraperError

    log = structlog.get_logger()
    META_API_URL = "https://graph.facebook.com/v25.0/ads_archive"

    class MetaAdsScraper:
        async def fetch_ads(self, product_title: str) -> list[dict]:
            token = os.getenv("META_ACCESS_TOKEN", "").strip()
            if not token:
                log.warning("meta_ads.skipped", reason="no_META_ACCESS_TOKEN")
                return []

            params = {
                "access_token": token,
                "search_terms": product_title[:100],
                "ad_reached_countries": "BR",   # OBRIGATÓRIO
                "ad_active_status": "ACTIVE",
                "ad_type": "ALL",
                "fields": "page_name,ad_snapshot_url,ad_creative_bodies,ad_delivery_start_time",
                "limit": 25,
            }
            async with httpx.AsyncClient(timeout=30.0) as client:
                try:
                    resp = await client.get(META_API_URL, params=params)
                    resp.raise_for_status()
                    return resp.json().get("data", [])
                except httpx.HTTPStatusError as e:
                    raise ScraperError(f"Meta API error: {e.response.status_code}") from e
    ```

    ```python
    from bs4 import BeautifulSoup
    from ..base_scraper import BaseScraper
    from ..exceptions import ScraperError
    import structlog

    log = structlog.get_logger()

    # Plataformas com reviews nativos e URL pattern
    NATIVE_REVIEW_PLATFORMS = {"hotmart", "clickbank", "kiwify"}

    class ReviewsScraper(BaseScraper):
        async def collect(self, platform: str, product_url: str, product_title: str = "") -> list[dict]:
            """Coleta reviews. Falha graciosa — nunca propaga exceção."""
            try:
                if platform in NATIVE_REVIEW_PLATFORMS:
                    return await self._collect_native(platform, product_url)
                else:
                    return await self._google_fallback(product_title)
            except ScraperError as e:
                log.warning("reviews.collect_failed", platform=platform, error=str(e))
                return []

        async def _collect_native(self, platform: str, url: str) -> list[dict]:
            html = await self.fetch(url)
            soup = BeautifulSoup(html, "lxml")
            reviews = []
            # Parser genérico: procurar por elementos de review (rating + texto)
            # Hotmart usa data-rating ou class com "star"; adaptar por plataforma
            for el in soup.select("[data-rating], .rating, .review-rating"):
                try:
                    rating = float(el.get("data-rating") or el.get_text(strip=True) or 0)
                    text_el = el.find_next(class_=lambda c: c and "text" in c.lower())
                    text = text_el.get_text(strip=True) if text_el else ""
                    if text:
                        reviews.append({
                            "text": text,
                            "valence": "positive" if rating >= 4.0 else "negative",
                            "rating": rating,
                            "source": platform,
                        })
                except (ValueError, AttributeError):
                    continue
            return reviews

        async def _google_fallback(self, product_title: str) -> list[dict]:
            if not product_title:
                return []
            query = f'"{product_title}" review'
            url = f"https://www.google.com/search?q={query.replace(' ', '+')}"
            try:
                html = await self.fetch(url)
                soup = BeautifulSoup(html, "lxml")
                # Extrair snippets de resultado de busca como reviews informais
                reviews = []
                for snippet in soup.select(".VwiC3b, .st")[:10]:
                    text = snippet.get_text(strip=True)
                    if len(text) > 20:
                        reviews.append({
                            "text": text,
                            "valence": "positive",  # sem rating → assumir positivo
                            "rating": None,
                            "source": "google",
                        })
                return reviews
            except ScraperError:
                return []
    ```
