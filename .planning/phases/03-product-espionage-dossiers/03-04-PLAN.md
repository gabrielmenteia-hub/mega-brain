    ```python
    import json, os
    from pathlib import Path
    from anthropic import AsyncAnthropic, APIConnectionError, RateLimitError
    from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type
    from ..spies.completeness_gate import SpyData
    import structlog

    log = structlog.get_logger()

    class CopyAnalysisError(Exception): ...

    _PROMPT_PATH = Path(__file__).parent.parent / "prompts" / "copy_analyzer.md"

    async def analyze_copy(data: SpyData) -> dict:
        """Etapa 1: analisa copy e identifica framework."""
        if not data.copy_text:
            raise CopyAnalysisError("copy_text ausente — gate deve ter bloqueado antes")

        result_text = await _call_llm_with_retry(data)
        try:
            return json.loads(result_text)
        except json.JSONDecodeError as e:
            raise CopyAnalysisError(f"LLM retornou não-JSON: {result_text[:200]}") from e

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=30),
        retry=retry_if_exception_type((APIConnectionError, RateLimitError)),
        reraise=True,
    )
    async def _call_llm_with_retry(data: SpyData) -> str:
        system_prompt = _PROMPT_PATH.read_text(encoding="utf-8")
        content = _build_user_content(data)
        client = AsyncAnthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
        response = await client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=2048,
            system=system_prompt,
            messages=[{"role": "user", "content": content}],
        )
        return response.content[0].text

    def _build_user_content(data: SpyData) -> str:
        parts = [f"# COPY DA PÁGINA DE VENDAS\n{data.copy_text}"]
        if data.offer_data:
            parts.append(f"\n# ESTRUTURA DA OFERTA\n{json.dumps(data.offer_data, ensure_ascii=False)}")
        if data.ads:
            parts.append(f"\n# ANÚNCIOS ATIVOS ({len(data.ads)})\n{json.dumps(data.ads[:5], ensure_ascii=False)}")
        if data.reviews:
            parts.append(f"\n# REVIEWS ({len(data.reviews)} coletados)\n{json.dumps(data.reviews[:20], ensure_ascii=False)}")
        return "\n".join(parts)
    ```

    Criar mis/intelligence/dossier_generator.py:
    ```python
    import json, os
    from datetime import datetime
    from pathlib import Path
    from anthropic import AsyncAnthropic, APIConnectionError, RateLimitError
    from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type
    from ..spies.completeness_gate import SpyData
    from ..db import get_db
    import structlog

    log = structlog.get_logger()

    class DossierGenerationError(Exception): ...

    _PROMPT_PATH = Path(__file__).parent.parent / "prompts" / "dossier_generator.md"

    async def generate_dossier(
        data: SpyData,
        copy_analysis: dict,
        dossier_id: int,
        db_path: str,
    ) -> dict:
        """Etapa 2: gera dossiê completo. Registra chamada LLM no banco."""
        content = _build_user_content(data, copy_analysis)
        system_prompt = _PROMPT_PATH.read_text(encoding="utf-8")

        client = AsyncAnthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
        response = await _call_llm_with_retry(client, system_prompt, content)

        # Registrar tokens e custo
        cost_usd = (response.usage.input_tokens * 0.000003 +
                    response.usage.output_tokens * 0.000015)
        db = get_db(db_path)
        db["llm_calls"].insert({
            "dossier_id": dossier_id,
            "model": "claude-sonnet-4-6",
            "stage": "dossier_generator",
            "input_tokens": response.usage.input_tokens,
            "output_tokens": response.usage.output_tokens,
            "cost_usd": round(cost_usd, 6),
            "created_at": datetime.utcnow().isoformat(),
        })

        try:
            return json.loads(response.content[0].text)
        except json.JSONDecodeError as e:
            raise DossierGenerationError(f"LLM retornou não-JSON: {response.content[0].text[:200]}") from e

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=30),
        retry=retry_if_exception_type((APIConnectionError, RateLimitError)),
        reraise=True,
    )
    async def _call_llm_with_retry(client, system_prompt, content):
        return await client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=4096,
            system=system_prompt,
            messages=[{"role": "user", "content": content}],
        )

    def _build_user_content(data: SpyData, copy_analysis: dict) -> str:
        parts = [
            f"# ANÁLISE DE COPY\n{json.dumps(copy_analysis, ensure_ascii=False)}",
            f"\n# COPY ORIGINAL\n{data.copy_text or ''}",
        ]
        if data.offer_data:
            parts.append(f"\n# OFERTA\n{json.dumps(data.offer_data, ensure_ascii=False)}")
        if data.reviews:
            parts.append(f"\n# REVIEWS ({len(data.reviews)})\n{json.dumps(data.reviews[:30], ensure_ascii=False)}")
        if data.ads:
            parts.append(f"\n# ANÚNCIOS ({len(data.ads)})\n{json.dumps(data.ads[:10], ensure_ascii=False)}")
        return "\n".join(parts)
    ```
